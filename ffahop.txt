-- Rivals FFA Server Hopper 

local TeleportService = game:GetService("TeleportService")
local Players = game:GetService("Players")
local HttpService = game:GetService("HttpService")
local RunService = game:GetService("RunService")

local player = Players.LocalPlayer
local placeId = game.PlaceId

local HOP_ENABLED = true   -- Always ON

-- ─── CONFIGURATION FOR PLAYER COUNT ───
local MIN_PLAYER_COUNT = 6 -- Automatically hops ONLY if the server falls below 8 players

local lastHopTime = tick()
local hopCount = 0

local function isServerGood(server)
    if not server or not server.id or server.id == game.JobId then
        return false
    end
    if server.playing >= server.maxPlayers * 0.88 then
        return false
    end
    return true
end

local function ServerHop()
    if not HOP_ENABLED then return end
    if tick() - lastHopTime < 8 then return end

    hopCount += 1
    print("🔄 Hop attempt #" .. hopCount)

    task.wait(0.7 + math.random() * 1.6)

    local success, response = pcall(function()
        local url = "https://games.roblox.com/v1/games/" .. placeId .. "/servers/Public?sortOrder=Desc&limit=100"
        return HttpService:JSONDecode(game:HttpGet(url, true))
    end)

    if not success or not response or not response.data then
        task.wait(4 + math.random() * 4)
        return
    end

    local validServers = {}
    for _, s in ipairs(response.data) do
        if isServerGood(s) then
            table.insert(validServers, s)
        end
    end

    if #validServers == 0 then
        task.wait(5 + math.random() * 5)
        return
    end

    -- Shuffle
    for i = #validServers, 2, -1 do
        local j = math.random(i)
        validServers[i], validServers[j] = validServers[j], validServers[i]
    end

    local target = validServers[math.random(1, math.min(6, #validServers))]

    print("✅ Hopping → " .. target.playing .. "/" .. target.maxPlayers .. " players")

    lastHopTime = tick()
    task.wait(0.5 + math.random() * 1.3)

    -- ─── AUTO EXECUTE ON TELEPORT LOGIC ───
    -- Registers your custom GitRequests sequence to execute automatically inside the next server
    local queue = queue_on_teleport or (syn and syn.queue_on_teleport)
    if queue then
        queue([[
            task.wait(3)
            pcall(function()
                local GitRequests = loadstring(game:HttpGet('https://raw.githubusercontent.com/csgofever/Roblox-GitRequests/refs/heads/main/GitRequests.lua'))()
                local Repo = GitRequests.Repo("csgofever", "Modules")
                loadstring(Repo:getFileContent("ffahop.txt"))()
            end)
        ]])
    end

    TeleportService:TeleportToPlaceInstance(placeId, target.id, player)
end

-- ─── PLAYER COUNT MONITOR LOOP ───
-- Checks every 3 seconds. It will ONLY hop if the population is below the threshold.
task.spawn(function()
    print("🚀 Rivals FFA Server Hopper Loaded")
    print("🛡️ Monitoring Active. Will only hop if players fall below: " .. MIN_PLAYER_COUNT)
    
    while task.wait(3) do
        local currentPlayers = #Players:GetPlayers()
        if HOP_ENABLED and currentPlayers < MIN_PLAYER_COUNT then
            print("⚠️ Low player count detected (" .. currentPlayers .. "/" .. MIN_PLAYER_COUNT .. "). Triggering hop...")
            ServerHop()
        end
    end
end)

-- Light anti-detection
RunService.Heartbeat:Connect(function()
    if math.random() < 0.01 then
        -- Fake activity
    end
end)

-- ─── AUTO EXECUTE VERIFICATION PRINT ───
print("✨ [AUTO-EXECUTE SUCCESS] Script has successfully loaded via teleport queue!")
